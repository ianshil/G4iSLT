(* Now we can prove that contraction is admissible. *)

Theorem G4iSLT_ctr_L : forall (m k: nat) s
        (D0 : derrec G4iSLT_rules (fun _ => False) s),
        k = (derrec_height D0) ->
          (forall sc A,
          (m = weight_form A) ->
          (@ctr_L A s sc) ->
          derrec G4iSLT_rules (fun _ => False) sc).
